n = INT ( INPUT ( " Enter the number of elements: ") ) .
myList = [ " a "," a " ," a " ," a " ," a " ," a " ," a " ," a " ," a " ] .
j = 0 .
counter = 0 .
i = 0 .

WHILE j < n:
    currentNumber = INT ( INPUT ( " Enter the number of elements less than 20: " ) ) .
    myList[j] = currentNumber .
    j = j + 1 .


PRINT ( " The number of elements divisible by 5: " )
WHILE i < n:
    IF myList[i] % 5 == 0:
        counter = counter + 1 .
    i = i + 1.
PRINT ( counter ) .
